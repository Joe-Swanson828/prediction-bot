# REQUIRES: Python 3.13 (use .venv created with python3.13)
#
# INSTALLATION (run from project root):
#   python3.13 -m venv .venv
#   source .venv/bin/activate
#   pip install web3==6.14.0 --no-deps
#   pip install eth-abi eth-account "eth-hash[pycryptodome]" eth-typing \
#               eth-utils hexbytes jsonschema "lru-dict==1.2.0" \
#               pyunormalize requests websockets aiohttp protobuf
#   pip install py-clob-client
#   pip install -r requirements.txt
#
# WHY --no-deps for web3:
#   web3==6.14.0 pins ckzg to an old version with no Python 3.13 wheels.
#   ckzg is only needed for Ethereum blob transactions (EIP-4844), which
#   Polymarket/Polygon never uses. py-clob-client installs ckzg 2.x which
#   does have Python 3.13 wheels and is compatible.
#   Never upgrade web3 to 7.x — py-clob-client requires 6.x.
#
# AFTER INSTALL: Remove web3's broken pytest plugin entry point:
#   sed -i '' '/pytest_ethereum/d' \
#     .venv/lib/python3.13/site-packages/web3-6.14.0.dist-info/entry_points.txt
#   (web3's pytest helpers require old eth-typing; we don't use them)
# ============================================================
